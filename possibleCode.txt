for(int neighbourA : neighboursA) {
                int payoffNeighbourA = payoffs[neighbourA][idB];
                if(payoffNeighbourA > payoffA) {
                    int prevID = idA;
                    
                    idA = neighbourA;
                    payoffA = payoffNeighbourA;

                    changedA = true;

                    System.out.println("Firm A: Changed from " + cities.get(prevID).getName() + " to " + cities.get(idA).getName());
                }
            }

            for(int neighbourB : neighboursB) {
                int payoffNeighbourB = payoffs[idB][neighbourB];
                if(payoffNeighbourB > payoffB) {
                    int prevID = idB;

                    idB = neighbourB;
                    payoffB = payoffNeighbourB;

                    changedB = true;

                    System.out.println("Firm B: Changed from " + cities.get(prevID).getName() + " to " + cities.get(idB).getName());
                }
            }

            if(!changedA) System.out.println("Firm A stays at " + cities.get(idA).getName() + " for this round.");
            if(!changedB) System.out.println("Firm B stays at " + cities.get(idB).getName() + " for this round.");
